import numpy as np 
import os
import cv2

file_path = os.path.dirname(os.path.realpath(__file__)) 
os.chdir(file_path)

series = os.listdir('../../Data_Memoir/Animated')
all_images = []

for srs in series:
    list_ = os.listdir('../../Data_Memoir/Animated/' + srs + '/Frames')
    all_images += [os.path.join('../../Data_Memoir/Animated/' + srs + '/Frames', i) for i in list_]

all_images = sorted(all_images)
# print(all_images)

total_num_images = len(all_images)
batch_size = 20
n = total_num_images // batch_size


[docs]def batch(image_names):
    img = []
    for nm in image_names:
        gh = cv2.imread(nm)
        img.append(cv2.resize(gh, (100, 100)))
    return np.array(img, dtype=np.float32)



means = []
start = 0
end = batch_size
for i in range(n):
    i_names = all_images[start: end]
    img_batch = batch(i_names)
    m = np.mean(img_batch, axis=(0, 1, 2))
    means.append(m)
    start += batch_size
    end += batch_size

means = np.array(means)
print(np.sum(means, axis=0) / n)
